10// Validation Functions
11function validateName($name) {
12 $name = trim($name);
13 if (empty($name)) return "Name cannot be empty.";
14 if (strlen($name) < 2) return "Name too short.";
15 if (!preg_match("/^[a-zA-Z ]+$/", $name)) return "Letters only.";
16 return "";
17}
18function validateEmail($email) {
19 $email = trim($email);
20 if (empty($email)) return "Email cannot be empty.";
21 if (!filter_var($email, FILTER_VALIDATE_EMAIL)) return "Invalid email.";
22 return "";
23}
24function validateFeedback($feedback) {
25 $feedback = trim($feedback);
26 if (empty($feedback)) return "Feedback cannot be empty.";
27 if (strlen($feedback) < 10) return "Feedback too short.";
28 return "";
29}
30function validateRating($rating) {
31 if (!is_numeric($rating) || $rating < 1 || $rating > 5) return "Rating: 1-5 only.";
32 return "";
33}
34
35// Store Feedback to File (Append Mode)
36function storeFeedback($file, $name, $email, $feedback, $rating) {
37 $timestamp = date('d-m-Y H:i:s');
38 $entry = "--------------------------------------------\n";
39 $entry .= "Date : $timestamp\n";
40 $entry .= "Name : $name\n";
41 $entry .= "Email : $email\n";
42 $entry .= "Rating : $rating/5\n";
43 $entry .= "Feedback : $feedback\n";
44 $entry .= "--------------------------------------------\n\n";
45 file_put_contents($file, $entry, FILE_APPEND | LOCK_EX);
46}
47
48// Display All Stored Feedback
49function displayFeedback($file) {
50 if (!file_exists($file) || filesize($file) === 0) {
51 echo " No feedback records found.\n"; return;
52 }
53 echo file_get_contents($file);
54}
