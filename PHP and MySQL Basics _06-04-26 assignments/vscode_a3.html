16// Load All Expenses into Array
17function loadExpenses($file) {
18 $expenses = [];
19 if (!file_exists($file)) return $expenses;
20 $lines = file($file, FILE_IGNORE_NEW_LINES | FILE_SKIP_EMPTY_LINES);
21 foreach ($lines as $line) {
22 $parts = explode('|', $line);
23 if (count($parts) === 3) {
24 $expenses[] = ["date"=>$parts[0],"category"=>$parts[1],
25 "amount"=>(float)$parts[2]];
26 }
27 }
28 return $expenses;
29}
